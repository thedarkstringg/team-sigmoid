\section{Motivation}
\begin{frame}{Motivation \& Central Question}
\begin{block}{Central Question}
  \textit{Under what conditions does boosting outperform bagging,\\
  and vice versa --- and \textbf{why}?}
\end{block}

\bigskip
\begin{columns}
\column{0.5\textwidth}
\textbf{Boosting (AdaBoost / SAMME)}
\begin{itemize}
  \item Re-weights hard examples
  \item Reduces bias sequentially
  \item Sensitive to noise
\end{itemize}
\column{0.5\textwidth}
\textbf{Bagging (Random Forest)}
\begin{itemize}
  \item Bootstrap aggregation
  \item Reduces variance in parallel
  \item Robust to noise
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Module 2 -- AdaBoost (SAMME)}
For $m = 1 \ldots M$:
\begin{enumerate}
  \item Fit stump $h_m$ with weights $\mathbf{w}^{(m)}$
  \item $\epsilon_m = \dfrac{\sum_i w_i^{(m)} \mathbf{1}[h_m(x_i)\neq y_i]}{\sum_i w_i^{(m)}}$
  \item $\alpha_m = \ln\dfrac{1-\epsilon_m}{\epsilon_m} + \ln(K-1)$
  \item $w_i^{(m+1)} \propto w_i^{(m)} \exp\!\bigl(\alpha_m \mathbf{1}[h_m(x_i)\neq y_i]\bigr)$
\end{enumerate}
\vfill
Final: $\hat{y} = \arg\max_k \sum_m \alpha_m \mathbf{1}[h_m(x)=k]$
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{Module 3 -- Random Forest}
\begin{itemize}
  \item $T$ trees on bootstrap samples $\mathcal{D}_t$ (with replacement)
  \item Each split selects $\lfloor\sqrt{p}\rfloor$ random features
  \item Majority vote for prediction; averaged probabilities for \texttt{predict\_proba}
  \item OOB score computed from out-of-bag samples
  \item Parallelism via \texttt{multiprocessing.Pool} (\texttt{n\_jobs})
\end{itemize}
\end{frame}

\section{Results \& Conclusion}
\begin{frame}{Key Findings}
\begin{block}{When does Boosting win?}
  Clean, balanced, binary datasets with enough rounds -- lower bias than RF.
\end{block}
\begin{block}{When does Bagging win?}
  Noisy labels, class imbalance, high dimensionality -- lower variance, better robustness.
\end{block}
\begin{block}{Unsupervised insight}
  PCA clusters align with class boundaries on clean data; DBSCAN reveals non-convex
  geometry that K-Means misses.
\end{block}
\end{frame}
